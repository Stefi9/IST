@echo off
SETLOCAL EnableDelayedExpansion
setlocal

:: IST Is a 64-bit program...
if /i "%PROCESSOR_ARCHITECTURE%"=="x86" if not defined PROCESSOR_ARCHITEW6432 (
    echo SYSERROR: IST is Unsupported on your 32-bit Opreating System
    echo Because of that, IST cannot be used.
    echo ERROR-CODE: 0x32
    pause
    exit /b
)
:: So no 32-bit user can use it (will just crash)

if /i "%~1"=="--EXEC-UNINSTALL" GOTO :PROCEED_UNINSTALL

:: Variables

:: Version
set "version=1.6"
set "ISTNCDV=1.3"

:: The directory of IST
set "IST-DIR=%~dp0"

:: If block (for PATH users)
if "%IST-DIR:~-1%"=="\" set "IST-DIR=%IST-DIR:~0,-1%"

:: The type of IST
if exist "%PROGRAMFILES%\IST\IST.bat" (
        set "type=install"
        set "ISTPGF=%PROGRAMFILES%\IST"
)
if not exist "%PROGRAMFILES%\IST\IST.bat" (
        set "type=portable"
        set "ISTPGF=N/A"
)

:: Administrator rights?
net session >nul 2>&1
if %errorLevel% == 0 (
    set "ADMIN_STATUS=1"
    set "ADMIN_LABEL=[ADMIN]"
) else (
    set "ADMIN_STATUS=0"
    set "ADMIN_LABEL=[USER]"
)

:: Title Variable
set "TitVar=IST %version% %ADMIN_LABEL%"

:: Lines of Code
set "line_count=0"
for /f %%a in ('find /c /v "" ^< "%~f0"') do set "line_count=%%a"

:: Title
title %TitVar%

:: Banner
for /f "tokens=2 delims=:" %%a in ('mode con ^| findstr Columns') do set /a "cols=%%a-2"
set "line="
for /l %%i in (1,1,%cols%) do set "line=!line!="
set /a "padding=(cols - 35) / 2"
set "spaces="
for /l %%i in (1,1,%padding%) do set "spaces=!spaces! "

:: cmd_count
set "cmd_count=52"
set "Ncmd_count=8"

:: YOLO through DeadZone
set "YOLO=0"
for %%x in (%*) do if /i "%%x"=="--yolo" set "YOLO=1"

:: Log Maintenance: Prevent infinite growth
if exist "%PROGRAMFILES%\IST\log-files" (
    :: If Action log exists and is > 500KB, wipe it
    powershell -NoProfile -Command "$path='%PROGRAMFILES%\IST\log-files\Actions.txt'; if(Test-Path $path){ if((Get-Item $path).Length -gt 500KB){ Clear-Content $path } }"
)


:: Check the first thing the user typed (%1)

:: Checks the internet
IF "%1"=="-check-network"            GOTO NETWORK
:: Shows your IPv4 address
IF "%1"=="-ip"                       GOTO IP
:: Show the DNS of a domain
IF "%1"=="-dns"                      GOTO DNS
:: Pings server chosen
IF "%1"=="-ping"                     GOTO PING
:: Clears the cache for the DNS on your computer to maybe fix your network issues
IF "%1"=="-flushdns"                 GOTO FLUSH
:: Shows information about internet in general on your device
IF "%1"=="-info"                     GOTO INFO
:: Shows the mac address
IF "%1"=="-mac"                      GOTO MAC
:: What is the set name to make the network recognize this device
IF "%1"=="-hostname"                 GOTO HOSTNAME
:: Shows how long the network card was turned on
IF "%1"=="-listend-ports"            GOTO LISPORT
:: Shows what apps use the internet right now
IF "%1"=="-apps"                     GOTO APPS
:: Shows you a refreshing view, of how much data is being sent and received every 3 seconds
IF "%1"=="-monitor"                  GOTO MONITOR
:: Maximum reachable speed
IF "%1"=="-max-speed"                GOTO MAXSPEED
:: Show every connected network on the windows device
IF "%1"=="-list-connected-nets"      GOTO LCN
:: Shows WiFi Password
IF "%1"=="-list-pass"                GOTO PASS
:: Shows WiFi Info of a specified WiFi network
IF "%1"=="-list-wifi-info"           GOTO LWINFO
:: Shows what other IPs/URLs a website needs before you can use it
IF "%1"=="-trace"                    GOTO TRACE
:: Shows Currently active IPs
IF "%1"=="-active-ips"               GOTO ACTIVE
:: Shows DNS cache that built up by the internet
IF "%1"=="-dns-cache-history"        GOTO HISTORY
:: Installes Drivers
IF "%1"=="-install-drivers"          GOTO DRIVERS
:: Lists how many bytes were Recieved and Sent
IF "%1"=="-usage"                    GOTO USAGE
:: Shows Status for every network adapter
IF "%1"=="-adapter-status"           GOTO ADAPTERSTATUS
:: Shows the Gateway of your Internet Connection
IF "%1"=="-gateway"                  GOTO GATE
:: Initial Setup wizard
if /i "%~1"=="--install" (
    :: 1. Check if already installed
    if exist "%PROGRAMFILES%\IST\IST.bat" (
        echo ERROR: IST is already installed.
        echo Use '-uninstall' to remove.
        echo ERROR-CODE: 0x1
        GOTO :EOF
    )

    :: 2. Check for Admin Rights (Crucial for C:\Program Files)
    net session >nul 2>&1
    if %errorlevel% neq 0 (
        echo ERROR: Installation requires Administrator privileges.
        echo Please right-click and 'Run as Administrator'.
        echo ERROR-CODE 0x2
        GOTO :EOF
    )

    :: 3. If both checks pass, proceed to the actual install logic
    GOTO :SETUP
)

:: Changes to the wanted DNS
IF "%1"=="-set-dns"                  GOTO SETDNS
:: Shows all Log Files
IF "%1"=="-show-log"                 GOTO LOG
:: shows if IST hasw updates
IF "%1"=="--update"                  GOTO UPDATE
:: Optimizes the windows device
IF "%1"=="-optimize"                 GOTO OPTIMIZE
:: Shows the Type of IST you are using (portable or installed)
IF "%1"=="-type"                     GOTO TYPE
:: Exports all Logs to the Desktop
IF "%1"=="-export-log"               GOTO EXPORTLOG
:: Restarts the Terminal
IF "%1"=="-restart"                  GOTO RESTART
:: Reinstalls IST
if /i "%~1"=="--reinstall" (
    :: 1. Check if it's even installed
    if not exist "%PROGRAMFILES%\IST\IST.bat" (
        echo ERROR: IST is not installed in Program Files.
        echo There is nothing to reinstall.
        echo ERROR-CODE: 0x01
        GOTO :EOF
    )

    :: 2. Check for Admin (Crucial for deleting from C:\Program Files)
    net session >nul 2>&1
    if %errorlevel% neq 0 (
        echo ERROR: Reinstallation REQUIRES Administrator privileges.
        echo ERROR-CODE: 0x2
        GOTO :EOF
    )

    :: 3. If it exists and we have power, go to the logic
    GOTO :REINSTALL
)

:: See The time, when IST was installed as local program
IF "%1"=="-installed-time"           GOTO INTTIME
:: Is the Internet Working?
IF "%1"=="-check-use"                GOTO CHECKUSE
:: Shows Connection Type
IF "%1"=="-connection"               GOTO CONNECTION
:: Lists All Logs to IST
IF "%1"=="-show-log-amount"          GOTO LOGAM
:: Shows What is new to IST
IF "%1"=="--whats-new"               GOTO WN
:: Shows Information of a given Error Code
IF "%1"=="--error-codes"             GOTO EC
:: TUI
IF "%1"=="--tui"                     GOTO CHECK_INTEGRITY
:: Checks the Proxy
IF "%1"=="-check-proxy"              GOTO CHECKPROXY
:: Coverty Bits to Bytes
IF "%1"=="-convert-bits"             GOTO CONVERTBYTES
:: Exports the Configs
IF "%1"=="-export-configs"           GOTO EXPORTCONFS

:: Others
:: Help centre
IF "%1"==""                          GOTO USE
:: Also help centre
IF "%1"=="--help"                    GOTO HELP
:: Shows version
IF "%1"=="--version"                 GOTO VERSION
:: ???
IF "%1"=="--123456789-123456789"     GOTO 123456789-123456789

:: DEAD ZONE
:: Makes a new IP address
IF "%1"=="--DeadZone-renew"          GOTO DEADRENEW
:: Removes The IST network configuration driver
IF "%1"=="--DeadZone-remove-driver"  GOTO DEADRMDRIVER
:: Resets WinSock (WARNING: it is not a sock)
IF "%1"=="--DeadZone-rmwinsock"      GOTO DEADRMSOCK
:: Makes a new IP address
IF "%1"=="--DeadZone-reset-firewall" GOTO DEADFIRE
:: Removes The IST network configuration driver
IF "%1"=="--DeadZone-ipv6-off"       GOTO DEADIPV6
:: Resets WinSock (WARNING: it is not a sock)
IF "%1"=="--DeadZone-set-dns"        GOTO DEADDNS
:: Uninstalls IST
if /i "%~1"=="--DeadZone-uninstall" (
    :: 1. Check if it's even installed
    if not exist "%PROGRAMFILES%\IST\IST.bat" (
        echo ERROR: IST is not installed in Program Files.
        echo There is nothing to uninstall.
        echo ERROR-CODE: 0x001
        GOTO :EOF
    )

    :: 2. Check for Admin (Crucial for deleting from C:\Program Files)
    net session >nul 2>&1
    if %errorlevel% neq 0 (
        echo ERROR: Uninstallation requires Administrator privileges.
        echo ERROR-CODE: 0x2
        GOTO :EOF
    )

    :: 3. If it exists and we have power, go to the logic
    GOTO :DEADUNINSTALL
)
:: Resets ARP
IF "%1"=="--deadzone-reset-arp"      GOTO DEADRSARP
:: Goes to the DeadZone Version of TUI
IF "%1"=="--deadzone-tui"            GOTO TUI-DEAD



:: For everything else, that is not recognized
IF NOT "%1"==""                      GOTO USE

:NETWORK
:: 1. Fast check: See if the command even exists
where speedtest >nul 2>&1
if %errorlevel% neq 0 (
    echo ERROR: Speedtest CLI not detected. Auto-installing Network Configuration driver...
    echo ERROR-CODE: 0x908

    :: 2. The install (only happens if 'where' fails)
    winget install --id Ookla.Speedtest.CLI -e --silent --accept-source-agreements --accept-package-agreements >nul 2>&1

    :: 3. Refresh the path so the script can see the new .exe immediately
    refreshenv >nul 2>&1 || set "PATH=%PATH%;%LOCALAPPDATA%\Microsoft\WinGet\Packages\Ookla.Speedtest.CLI_Microsoft.Winget.Source_8wekyb3d8bbwe"
)

:: If 'where' succeeded (errorlevel 0), it skips straight to here!
echo Running Speed Test...
speedtest --accept-license --accept-gdpr
GOTO :EOF

:IP
:: This shows your local and public IP info
echo Your Local IP Information:
ipconfig | findstr "IPv4"
echo.
echo Your Public IP:

powershell -command "(Invoke-RestMethod -Uri 'https://api.ipify.org')"
GOTO :EOF

:DNS
if "%2"=="" (echo Please specify a domain ^(e.g., ist -dns google.com^) & exit /b)
echo Resolving IPs for %2...
nslookup %2 | findstr /i "address"
GOTO :EOF

:PING
:: 1. Ask who to ping
set "target="
set /p "target=Ping who? (Leave blank for google.com): "
if "%target%"=="" set "target=google.com"

:: 2. Ask if it should be continuous
set "ptype="
set /p "ptype=do you want to ping %target% continuously? (y/n)

echo.
if /i "%ptype%"=="y" (
    echo Pinging %target% continuously... Press CTRL+C to stop
    ping %target% -t
) else (
    echo Pinging %target%...
    ping %target%
)
GOTO :EOF

:FLUSH
echo Flushing DNS cache...
ipconfig /flushdns
GOTO :EOF

:INFO
echo Getting info...
getmac /v
GOTO :EOF

:MAC
echo Getting MAC address...
getmac
GOTO :EOF

:HOSTNAME
echo Showing the name of the windows device...
hostname
GOTO :EOF

:LISPORT
echo Finding Listening ports...
netstat -an | findstr "LISTENING"
GOTO :EOF

:APPS
:: 1. Check for Admin rights
net session >nul 2>&1
if %errorlevel% neq 0 (
    echo ERROR: '-apps' requires Administrator permission. Please try again
    echo ERROR-CODE: 0x2
    GOTO :EOF
)

:: 2. If Admin, run the command
echo Listing apps using the internet...
echo (This may take a few seconds)
netstat -b
GOTO :EOF

:MONITOR
echo To cancel, click CTRL + C
netstat -s 5
GOTO :EOF

:MAXSPEED
:: Detect the parent process
set "shell=cmd"
:: Note: %PPID% isn't a standard CMD variable, but your tasklist check works well
tasklist /fi "pid eq %PPID%" 2>nul | findstr /i "pwsh powershell" >nul && set "shell=powershell"

if "%shell%"=="powershell" (
    echo Running command...
    :: We escape the parentheses so CMD doesn't close the IF block early
    ^(Get-NetAdapter^).LinkSpeed
) else (
    echo Running command...
    powershell -command ^(Get-NetAdapter^).LinkSpeed"
)
GOTO :EOF

:LCN
echo Showing connected networks...
netsh wlan show profiles
GOTO :EOF

:PASS
if "%~2"=="" (
    echo Error: Please provide a network name.
    echo Usage: ist -wifi-pass "Network Name" (if the name has spaces, put in quotes)
    echo ERROR-CODE: 0x102
)

echo Searching for password...
:: We use /I to ignore case and look for "Key"
netsh wlan show profile name="%~2" key=clear
GOTO :EOF

:LWINFO
if "%~2"=="" (
    echo Error: Please provide a network name.
    echo Usage: ist -wifi-pass "Network Name" (if the name has spaces, put in quotes)
    echo ERROR-CODE: 0x102
)

netsh wlan show profile name="%~2" key=clear
GOTO :EOF

:TRACE
:: 1. Check if the second argument is empty
if "%~2"=="" (
    echo Error: No target provided.
    echo Usage: ist -trace [url/IP]
    echo ERROR-CODE: 0x102
    GOTO :EOF
)

echo Tracing route to %~2...
echo [TIP] Press Ctrl+C to cancel if it takes too long.
echo.

:: 2. Use -d to prevent it from hanging on IP resolution
:: and -h 10 to limit the search so it doesn't loop forever
tracert -d -h 15 %~2

if %errorlevel% neq 0 (
    echo.
    echo Trace failed. Check your internet or the URL spelling.
    echo ERROR-CODE: 0x102
)
GOTO :EOF

:ACTIVE
netstat -n | findstr "ESTABLISHED"
GOTO :EOF

:HISTORY
ipconfig /displaydns | findstr "Record Name"
GOTO :EOF

:DRIVERS
echo.
:: Fixed title syntax (no = needed)
title IST %version% - Drivers Installer
echo 1. Speedtest CLI (ist -check-network)
echo.
set /p "choice=Select drivers (A for All, Q to quit): "

if /i "%choice%"=="q" GOTO :EOF

:: 1. Show the "Starting" message first
echo.
echo Starting installation...

:: 2. Logic for the 'A' or '1' selection
if "%choice%"=="1" (
    :: Run the spinner for a few seconds so they see it, then install
    powershell -NoProfile -Command "$s='|','/','-','\'; for($i=0;$i -lt 20;$i++){ foreach($c in $s){ write-host -NoNewline (\"`rInstalling Speedtest CLI... $c\"); Start-Sleep -m 100 } }"
    winget install --id Ookla.Speedtest.CLI -e --silent --accept-source-agreements --accept-package-agreements
)

if /i "%choice%"=="A" (
    echo Installing All Drivers...
    :: We use a loop to install everything in the list
    for %%g in (Ookla.Speedtest.CLI) do (
        echo | set /p ="Installing %%g... "
        winget install --id %%g -e --silent --accept-source-agreements --accept-package-agreements >nul 2>&1
        echo [ OK ] Installed
    )
)

echo.
echo [ OK ] Task finished.
GOTO :EOF

:USAGE
echo Gathering Info...
netstat -e
GOTO :EOF

:ADAPTERSTATUS
echo Listing Status...
netsh int show interface
GOTO :EOF

:GATE
echo Showing...
ipconfig | findstr "Gateway"
GOTO :EOF

:SETUP
set "STtitle=IST: Internet Servicing Tool Full Setup wizard"
echo !line!
echo !spaces!!STtitle!
echo !line!

net session >nul 2>&1
if %errorlevel% neq 0 (
    echo ERROR: Setting IST fully up, REQUIRES Administrator privileges.
    echo Please Restart the terminal as Administrator.
    echo ERROR-CODE: 0x2
    GOTO :EOF
)

echo Starting Full Setup...

:: 1/5, ProgramFiles Add
echo [1/5] Installing IST as Local Program...
mkdir "%PROGRAMFILES%\IST"
mkdir "%PROGRAMFILES%\IST\log-files"
mkdir "%PROGRAMFILES%\IST\external-scripts"
:: Log Data [1/5]
echo [%date% %time%] Created Installation Folder "%PROGRAMFILES%\ist" >> "%PROGRAMFILES%\IST\log-files\Installation-log.txt"
echo [%date% %time%] Created Installation Folder "%PROGRAMFILES%\ist\log-files" >> "%PROGRAMFILES%\IST\log-files\Installation-log.txt"
echo [%date% %time%] Created Installation Folder "%PROGRAMFILES%\ist\external-scripts" >> "%PROGRAMFILES%\IST\log-files\Installation-log.txt"
echo [ OK ] Made Installation folders and files

:: 2/5, Drivers
echo [2/5] Installing Drivers (Auto-selecting 'A')...
:: Use 'echo a |' to feed the input into the call
echo a | call "%~f0" -install-drivers >nul
:: Log Data [2/10]
echo [%date% %time%] Installed Drivers >> "%PROGRAMFILES%\IST\log-files\Installation-log.txt"

:: 3/5, setting IST as PATH
echo [3/5] Setting IST to systems PATH...
powershell -Command "[Environment]::SetEnvironmentVariable('Path', [Environment]::GetEnvironmentVariable('Path', 'Machine') + ';%PROGRAMFILES%\IST', 'Machine')"
echo [ OK ] Set IST in Systems PATH.

:: 4/5, Flushing DNS, thus uptimizing the system
echo [4/5] Flushing DNS and optimizing...
call "%~f0" -flushdns >nul
echo [%date% %time%] Flushed the DNS >> "%PROGRAMFILES%\IST\log-files\Installation-log.txt"


:: 5/5, Moving IST
echo [5/5] Moving IST to ProgramFiles...
echo [ NOTICE ] while IST is moving, IST cannot do other commmands at the same time while moving
copy /y "%~f0" "%PROGRAMFILES%\IST\IST.bat" >nul
echo Installation Complete!
start "" "%PROGRAMFILES%\IST\IST.bat"
echo [%date% %time%] Moved IST into "%PROGRAMFILES%\IST" >> "%PROGRAMFILES%\IST\log-files\Installation-log.txt"

:: Finishing
echo Finishing Installation...
echo Installation Finished.

echo Restarting the Terminal
call "%~f0" -restart >nul
GOTO :EOF

:SETDNS
net session >nul 2>&1
if %errorlevel% neq 0 (
    echo ERROR: Setting a custom DNS, REQUIRES Administrator privileges.
    echo Please Restart the terminal as Administrator.
    echo ERROR-CODE: 0x2
    GOTO :EOF
)
echo presets:
echo ist -set-dns gaming
echo ist -set-dns stability
echo ist -set-dns google
echo ist -set-dns cloudflare
echo ist -set-dns security
echo ist -set-dns ad-protection

:: 1. Define the Presets (The "Database")
set "gaming=1.0.0.1"
set "stability=8.8.4.4"
set "google=8.8.8.8"
set "cloudflare=1.1.1.1"
set "security=9.9.9.9"
set "ad-protection=94.140.14.14"

:: 2. The Bridge: Point %primary% to the chosen preset
:: This uses 'indirect expansion' to grab the value of whatever the user typed
set "primary=!%~2!"

:: 3. Validation: Make sure the user didn't type a fake preset
if "%primary%"=="" (
    echo ERROR: '%~2' is not a valid preset.
    echo Use: gaming, stability, google, cloudflare, etc.
    echo ERROR-CODE: 0x102
    GOTO :EOF
)

:: 4. The Action
echo Applying %~2 DNS (%primary%)...
netsh interface ip set dns name="%adapter%" source=static address=%primary%
ipconfig /flushdns >nul
echo [ OK ] DNS updated successfully.
GOTO :EOF

:LOG
type "%ISTPGF%\log-files\Installation-log.txt"
GOTO :EOF

:UPDATE
:: [Version fetching logic stays the same]

if "%version%"=="%latest_version%" (
    echo [ OK ] You are up to date.
) else (
    echo Update available! - %latest_version%
    echo Downloading...

    :: Use clean variables without extra quotes
    set "zipURL=https://github.com/Stefi9/IST/releases/latest/download/IST.zip"
    set "destination=%USERPROFILE%\Downloads\update.zip"

    :: Fix: PowerShell now uses the CORRECT variable name ($destination)
    powershell -Command "Invoke-WebRequest -Uri '%zipURL%' -OutFile '%destination%'"

    echo Extracting...
    if not exist "%PROGRAMFILES%\IST" mkdir "%PROGRAMFILES%\IST"
    if not exist "%PROGRAMFILES%\IST\log-files" mkdir "%PROGRAMFILES%\IST\log-files"

    :: Fix: Changed %zipPath% to %destination% so PowerShell can find the file
    powershell -Command "Expand-Archive -Path '%destination%' -DestinationPath '%PROGRAMFILES%\IST' -Force"

    echo [ OK ] IST has been updated to %latest_version%.
)
GOTO :EOF

:OPTIMIZE
net session >nul 2>&1
if %errorlevel% neq 0 (
    echo ERROR: Optimizing (fully), REQUIRES Administrator privileges.
    echo Please Restart the terminal as Administrator.
    echo ERROR-CODE: 0x2
    GOTO :EOF
)
echo.
ipconfig /flushdns >nul
echo [%date% %time%] Flushed DNS >> "%PROGRAMFILES%\IST\log-files\Optimization-log.txt"
netsh int ipv6 set teredo disabled
echo [%date% %time%] Disabled IPv6 >> "%PROGRAMFILES%\IST\log-files\Optimization-log.txt"
netsh int ip set dns "Wi-Fi" static 8.8.8.8
echo [%date% %time%] Set DNS to Google's DNS >> "%PROGRAMFILES%\IST\log-files\Optimization-log.txt"
echo [ OK ] Device was Optmized successfully.
GOTO :EOF

:TYPE
echo Your IST version is %type%
GOTO :EOF

:EXPORTLOG
copy "%PROGRAMFILES%\IST\log-files" "%USERPROFILE%\Desktop\"
GOTO :EOF

:RESTART
start cmd.exe
exit

:REINSTALL
set "RTtitle=IST: Internet Servicing Tool Reinstallation"
echo !line!
echo !spaces!!RTtitle!
echo !line!

net session >nul 2>&1
if %errorlevel% neq 0 (
    echo ERROR: Reinstalling IST, REQUIRES Administrator privileges.
    echo Please Restart the terminal as Administrator.
    echo ERROR-CODE: 0x2
    GOTO :EOF
)
:: 1. Check if we are currently running from the folder we want to delete
if /i "%~dp0"=="%PROGRAMFILES%\IST\" (
    :: Copy the script to the Windows Temp folder
    copy /y "%~f0" "%temp%\IST_Updater.bat" >nul

    :: Run the copy from the Temp folder and tell it to skip back to the reinstall logic
    start "" "%temp%\IST_Updater.bat" --123456789-123456789

    :: Close this version immediately so the folder is no longer "in use"
    exit /b
)

:123456789-123456789
echo Now running from safe zone. Deleting old files...

:: 1/4, Keeping IST in Downloads as Portable Program
echo [1/4] Keeping IST in Downloads as a Portable Program...
copy /y "%~f0" "%USERPROFILE%\Downloads\IST_Portable.bat"
echo [ OK ] moved IST in downloads as a Portable Program

:: 2/5, Removing Files
echo [2/4] Removing Files...
rd /s /q "%PROGRAMFILES%\IST"
echo [ OK ] Removed Files.

:: 3/4, removing IST from PATH
echo [3/4] Remowing IST from PATH...
powershell -Command "$p=[Environment]::GetEnvironmentVariable('Path','Machine'); $p=$p -replace ';%PROGRAMFILES:\=\\%\\IST',''; [Environment]::SetEnvironmentVariable('Path',$p,'Machine')"

:: 4/4, Flushing DNS, thus uptimizing the system
echo [4/4] Flushing DNS and optimizing...
call "%~f0" -flushdns >nul
echo [ OK ] Flushing DNS finished.

:: Finishing
echo Finishing Uninstallation...
echo Uninstallation Finished.

:: 1/5, ProgramFiles Add
echo [1/5] Installing IST as Local Program...
mkdir "%PROGRAMFILES%\IST"
mkdir "%PROGRAMFILES%\IST\log-files"
mkdir "%PROGRAMFILES%\IST\external-scripts"
:: Log Data [1/5]
echo [%date% %time%] Created Installation Folder "%PROGRAMFILES%\ist" >> "%PROGRAMFILES%\IST\log-files\Installation-log.txt"
echo [%date% %time%] Created Installation Folder "%PROGRAMFILES%\ist\log-files" >> "%PROGRAMFILES%\IST\log-files\Installation-log.txt"
echo [%date% %time%] Created Installation Folder "%PROGRAMFILES%\ist\external-scripts" >> "%PROGRAMFILES%\IST\log-files\Installation-log.txt"
echo [ OK ] Made Installation folders and files

:: 2/5, Drivers
echo [2/5] Installing Drivers (Auto-selecting 'A')...
:: Use 'echo a |' to feed the input into the call
echo a | call "%~f0" -install-drivers >nul
:: Log Data [2/10]
echo [%date% %time%] Installed Drivers >> "%PROGRAMFILES%\IST\log-files\Installation-log.txt"

:: 3/5, setting IST as PATH
echo [3/5] Setting IST to systems PATH...
powershell -Command "[Environment]::SetEnvironmentVariable('Path', [Environment]::GetEnvironmentVariable('Path', 'Machine') + ';%PROGRAMFILES%\IST', 'Machine')"
echo [ OK ] Set IST in Systems PATH.

:: 4/5, Flushing DNS, thus uptimizing the system
echo [4/5] Flushing DNS and optimizing...
call "%~f0" -flushdns >nul
echo [%date% %time%] Flushed the DNS >> "%PROGRAMFILES%\IST\log-files\Installation-log.txt"


:: 5/5, Moving IST
echo [5/5] Moving IST to ProgramFiles...
echo [ NOTICE ] while IST is moving, IST cannot do other commmands at the same time while moving
copy /y "%~f0" "%PROGRAMFILES%\IST\IST.bat" >nul
echo Installation Complete!
start "" "%PROGRAMFILES%\IST\IST.bat"
echo [%date% %time%] Moved IST into "%PROGRAMFILES%\IST" >> "%PROGRAMFILES%\IST\log-files\Installation-log.txt"

:: Finishing
echo Finishing Installation...
echo Installation Finished.

echo Restarting the Terminal
call "%~f0" -restart >nul
GOTO :EOF
